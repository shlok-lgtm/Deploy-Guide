from app.database import fetch_one, fetch_all

print("=== TOTAL STATE ===")
tables = [
    ("Score history records", "SELECT COUNT(*) AS cnt FROM score_history"),
    ("Component readings", "SELECT COUNT(*) AS cnt FROM component_readings"),
    ("Assessment events", "SELECT COUNT(*) AS cnt FROM assessment_events"),
    ("Stablecoins scored", "SELECT COUNT(*) AS cnt FROM scores WHERE overall_score IS NOT NULL"),
    ("Protocols scored", "SELECT COUNT(DISTINCT protocol_slug) AS cnt FROM psi_scores"),
    ("Wallets discovered", "SELECT COUNT(*) AS cnt FROM wallet_graph.wallets"),

    ("Tracked USD", "SELECT COALESCE(SUM(total_stablecoin_value), 0) AS cnt FROM wallet_graph.wallet_risk_scores"),
]
total = 0
for label, sql in tables:
    try:
        row = fetch_one(sql)
        val = row["cnt"] if row else 0
        print(f"  {label}: {val:,.0f}")
        if label != "Tracked USD":
            total += int(val or 0)
    except Exception as e:
        print(f"  {label}: ERROR - {e}")
print(f"\n  TOTAL STATE RECORDS: {total:,}")
